set(NAZARA_SOURCE_HARDWAREINFO_DIRECTORY ${NAZARA_EXAMPLES_DIRECTORY}HardwareInfo)

if(${OS_RENDERER_SPECIFICATION} STREQUAL GLX)
	add_definitions(-DNAZARA_RENDERER_OPENGL)
endif()

		${NAZARA_INCLUDE_DIRECTORY}
		${NAZARA_EXTLIBS_DIRECTORY}/include/
	)

file(GLOB HARDWAREINFO_SOURCE
		${NAZARA_SOURCE_HARDWAREINFO_DIRECTORY}/*.cpp
	)

set(EXEC_NAME ${NAZARA_EXEC_NAME}HardwareInfo)

link_directories(${NAZARA_LIBRARY_DIRECTORY})

add_executable(${EXEC_NAME} ${HARDWAREINFO_SOURCE})

target_link_libraries(${NAZARA_EXEC_NAME}HardwareInfo
		${NAZARA_LIB_CORE}
		${NAZARA_LIB_GRAPHICS}
		${NAZARA_LIB_RENDERER}
		${NAZARA_LIB_UTILITY}
	)
