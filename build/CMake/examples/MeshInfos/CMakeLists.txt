project(MeshInfos)

set(NAZARA_SOURCE_MESHINFOS_DIRECTORY ${NAZARA_EXAMPLES_DIRECTORY}MeshInfos/)

include_directories(${NAZARA_INCLUDE_DIRECTORY})

file(GLOB MESHINFOS_SOURCE
		${NAZARA_SOURCE_MESHINFOS_DIRECTORY}*.cpp
	)

set(EXEC_NAME ${NAZARA_DEMO_NAME}MeshInfos)

link_directories(${NAZARA_LIBRARY_DIRECTORY})

set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${NAZARA_DEMO_OUTPUT_DIRECTORY})

add_executable(${EXEC_NAME} ${MESHINFOS_SOURCE})

target_link_libraries(${EXEC_NAME}
		${NAZARA_LIB_CORE}
		${NAZARA_LIB_UTILITY}
	)
