project("DopplerEffect")

set(NAZARA_SOURCE_DOPPLEREFFECT_DIRECTORY ${NAZARA_EXAMPLES_DIRECTORY}DopplerEffect/)

include_directories(${NAZARA_INCLUDE_DIRECTORY})

file(GLOB DOPPLER_SOURCE
		${NAZARA_SOURCE_DOPPLEREFFECT_DIRECTORY}*.cpp
	)

set(EXEC_NAME ${NAZARA_DEMO_NAME}DopplerEffect)

link_directories(${NAZARA_LIBRARY_DIRECTORY})

set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${NAZARA_DEMO_OUTPUT_DIRECTORY})

add_executable(${EXEC_NAME} ${DOPPLER_SOURCE})

target_link_libraries(${EXEC_NAME}
		${NAZARA_LIB_AUDIO}
		${NAZARA_LIB_CORE}
		${NAZARA_LIB_UTILITY}
	)
