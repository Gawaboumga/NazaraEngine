cmake_minimum_required (VERSION 2.8.11)
project (Nazara)

set (CMAKE_CXX_FLAGS  "${CMAKE_CXX_FLAGS} -std=c++14")

set (BUILD_SHARED_LIBS SHARED)
set (BUILD_SHARED_EXTLIBS SHARED)
set (NAZARA_EXTLIBS_EXTENSION "-d")
set (NAZARA_LIBRARY_EXTENSION "-d")
set (SDK_LIBRARY_EXTENSION "-d")

set (NAZARA_DIRECTORY "../../../../")

set (NAZARA_EXAMPLES_DIRECTORY ${NAZARA_DIRECTORY}examples/)
set (NAZARA_EXTLIBS_DIRECTORY ${NAZARA_DIRECTORY}extlibs/)
set (NAZARA_EXTLIBS_LIB_DIRECTORY ${NAZARA_EXTLIBS_DIRECTORY}lib/)
set (NAZARA_EXTLIBS_INCLUDE_DIRECTORY ${NAZARA_EXTLIBS_DIRECTORY}include/)
set (NAZARA_INCLUDE_DIRECTORY ${NAZARA_DIRECTORY}include/)
set (NAZARA_SOURCE_DIRECTORY ${NAZARA_DIRECTORY}src/)
set (NAZARA_LIBS_OUTPUT_DIRECTORY ${NAZARA_DIRECTORY}../lib/CMake/)
set (NAZARA_EXTLIBS_OUTPUT_DIRECTORY ${NAZARA_DIRECTORY}../extlibs/lib/CMake/)

set (NAZARA_DEMO_OUTPUT_DIRECTORY ${NAZARA_DIRECTORY}../examples/bin/)
set (NAZARA_DEMO_NAME "Demo")
set (NAZARA_TESTS_OUTPUT_DIRECTORY ${NAZARA_DIRECTORY}../tests/)
set (NAZARA_TESTS_DIRECTORY ${NAZARA_DIRECTORY}tests/)

set (SDK_INCLUDE_DIRECTORY ${NAZARA_DIRECTORY}SDK/include/)
set (SDK_SOURCE_DIRECTORY ${NAZARA_DIRECTORY}SDK/src/)

set (NAZARA_LIBRARY_NAME Nazara)
set (NAZARA_LIBRARY_DIRECTORY ${CMAKE_LIBRARY_OUTPUT_DIRECTORY})
set (NAZARA_LIB_AUDIO NazaraAudio-d)
set (NAZARA_LIB_CORE NazaraCore-d)
set (NAZARA_LIB_GRAPHICS NazaraGraphics-d)
set (NAZARA_LIB_LUA NazaraLua-d)
set (NAZARA_LIB_NOISE NazaraNoise-d)
set (NAZARA_LIB_NETWORK NazaraNetwork-d)
set (NAZARA_LIB_PHYSICS NazaraPhysics-d)
set (NAZARA_LIB_RENDERER NazaraRenderer-d)
set (NAZARA_LIB_UTILITY NazaraUtility-d)

set (NAZARA_LIB_SDK SDK-d)
set (NAZARA_LIB_SDKSERVER SDKServer-d)

add_subdirectory (extlibs)

add_subdirectory (modules)
add_subdirectory (tools)
add_subdirectory (examples)
