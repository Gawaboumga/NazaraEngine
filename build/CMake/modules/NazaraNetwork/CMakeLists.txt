project("NazaraNetwork")

if (UNIX)
	set (OS_NETWORK_SPECIFICATION "Posix")
elseif (WIN32)
	set (OS_NETWORK_SPECIFICATION "Win32")
else ()
	message(SEND_ERROR "No OS implementation")
endif ()

set(NAZARA_SOURCE_NETWORK_DIRECTORY ${NAZARA_SOURCE_DIRECTORY}Nazara/Network/)

include_directories(
		${NAZARA_INCLUDE_DIRECTORY}
		${NAZARA_SOURCE_DIRECTORY}
		${NAZARA_EXTLIBS_INCLUDE_DIRECTORY}
	)

file(GLOB NETWORK_SOURCE
		${NAZARA_SOURCE_NETWORK_DIRECTORY}Debug/*.cpp
		${NAZARA_SOURCE_NETWORK_DIRECTORY}${OS_NETWORK_SPECIFICATION}/*.cpp
		${NAZARA_SOURCE_NETWORK_DIRECTORY}*.cpp
	)

link_directories(${NAZARA_EXTLIBS_OUTPUT_DIRECTORY})

set(CMAKE_LIBRARY_OUTPUT_DIRECTORY ${NAZARA_LIBS_OUTPUT_DIRECTORY})

add_library(${NAZARA_LIBRARY_NAME}Network${NAZARA_LIBRARY_EXTENSION} ${BUILD_SHARED_LIBS} ${NETWORK_SOURCE})

target_link_libraries(${NAZARA_LIBRARY_NAME}Network${NAZARA_LIBRARY_EXTENSION})
if (UNIX)
	target_link_libraries(${NAZARA_LIBRARY_NAME}Network${NAZARA_LIBRARY_EXTENSION})
elseif (WIN32)
	target_link_libraries(${NAZARA_LIBRARY_NAME}Network${NAZARA_LIBRARY_EXTENSION}
			PUBLIC
				"ws2_32"
		)
else ()
	message(SEND_ERROR "No OS implementation")
endif ()
