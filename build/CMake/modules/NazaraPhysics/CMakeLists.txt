project("NazaraPhysics")

set(NAZARA_SOURCE_PHYSICS_DIRECTORY ${NAZARA_SOURCE_DIRECTORY}Nazara/Physics/)

include_directories(
		${NAZARA_INCLUDE_DIRECTORY}
		${NAZARA_EXTLIBS_DIRECTORY}/include/
	)

file(GLOB PHYSICS_SOURCE
		${NAZARA_SOURCE_PHYSICS_DIRECTORY}Debug/*.cpp
		${NAZARA_SOURCE_PHYSICS_DIRECTORY}*.cpp
	)

link_directories(${NAZARA_EXTLIBS_LIB_DIRECTORY})

set(CMAKE_LIBRARY_OUTPUT_DIRECTORY ${NAZARA_LIBS_OUTPUT_DIRECTORY})

add_library(${NAZARA_LIBRARY_NAME}Physics${NAZARA_LIBRARY_EXTENSION} ${BUILD_SHARED_LIBS} ${PHYSICS_SOURCE})

target_link_libraries(${NAZARA_LIBRARY_NAME}Physics${NAZARA_LIBRARY_EXTENSION}
		PRIVATE
			"Newton"
	)
