project("NazaraCore")

if (UNIX)
	set (OS_CORE_SPECIFICATION "Posix")
elseif (WIN32)
	set (OS_CORE_SPECIFICATION "Win32")
else ()
	message(SEND_ERROR "No OS implementation")
endif ()

set(NAZARA_SOURCE_CORE_DIRECTORY ${NAZARA_SOURCE_DIRECTORY}Nazara/Core/)

include_directories(
		${NAZARA_INCLUDE_DIRECTORY}
		${NAZARA_SOURCE_DIRECTORY}
		${NAZARA_EXTLIBS_INCLUDE_DIRECTORY}
	)

file(GLOB CORE_SOURCE
		${NAZARA_SOURCE_CORE_DIRECTORY}Debug/*.cpp
		${NAZARA_SOURCE_CORE_DIRECTORY}Hash/*.cpp
		${NAZARA_SOURCE_CORE_DIRECTORY}Hash/SHA/*.cpp
		${NAZARA_SOURCE_CORE_DIRECTORY}${OS_CORE_SPECIFICATION}/*.cpp
		${NAZARA_SOURCE_CORE_DIRECTORY}*.cpp
	)

set(CMAKE_LIBRARY_OUTPUT_DIRECTORY ${NAZARA_LIBS_OUTPUT_DIRECTORY})

add_library(${NAZARA_LIBRARY_NAME}Core${NAZARA_LIBRARY_EXTENSION} ${BUILD_SHARED_LIBS} ${CORE_SOURCE})

target_link_libraries(${NAZARA_LIBRARY_NAME}Core${NAZARA_LIBRARY_EXTENSION}
		"dl"
		"pthread"
	)
