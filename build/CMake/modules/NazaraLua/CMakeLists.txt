project("NazaraLua")

set(NAZARA_SOURCE_LUA_DIRECTORY ${NAZARA_SOURCE_DIRECTORY}Nazara/Lua/)

include_directories(
		${NAZARA_INCLUDE_DIRECTORY}
		${NAZARA_EXTLIBS_INCLUDE_DIRECTORY}
	)

file(GLOB LUA_SOURCE
		${NAZARA_SOURCE_LUA_DIRECTORY}Debug/*.cpp
		${NAZARA_SOURCE_LUA_DIRECTORY}*.cpp
	)

link_directories(${NAZARA_EXTLIBS_LIB_DIRECTORY})

set(CMAKE_LIBRARY_OUTPUT_DIRECTORY ${NAZARA_LIBS_OUTPUT_DIRECTORY})

add_library(${NAZARA_LIBRARY_NAME}Lua${NAZARA_LIBRARY_EXTENSION} ${BUILD_SHARED_LIBS} ${LUA_SOURCE})

target_link_libraries(${NAZARA_LIBRARY_NAME}Lua${NAZARA_LIBRARY_EXTENSION}
		PRIVATE
			"lua${NAZARA_EXTLIBS_EXTENSION}"
	)
